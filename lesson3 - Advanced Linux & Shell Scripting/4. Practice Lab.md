## Exercise 1: Install nginx

- Let remove nginx first `sudo apt remove nginx`

### Requirements
- Write shell script to
- Install nginx to the VM if nginx not yet installed
- Start nginx if it stopped
- enable nginx if disabled
- Check nginx is acctually running or not

### Instructions

```shell
#!/bin/bash

set -e  # Exit on any error

# Color codes for output
RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[1;33m'
NC='\033[0m' # No Color

# Function to print colored output
print_status() {
    echo -e "${GREEN}[INFO]${NC} $1"
}

print_warning() {
    echo -e "${YELLOW}[WARNING]${NC} $1"
}

print_error() {
    echo -e "${RED}[ERROR]${NC} $1"
}

# Function to check if nginx is installed
check_nginx_installed() {
    if command -v nginx > /dev/null 2>&1; then
        return 0  # nginx is installed
    else
        return 1  # nginx is not installed
    fi
}

# Function to install nginx
install_nginx() {
    print_status "Nginx not found. Installing nginx..."
    
    # Update package list
    print_status "Updating package list..."
    sudo apt update
    
    # Install nginx
    print_status "Installing nginx..."
    sudo apt install -y nginx
    
    if [ $? -eq 0 ]; then
        print_status "Nginx installed successfully!"
    else
        print_error "Failed to install nginx!"
        exit 1
    fi
}

# Function to check nginx service status
check_nginx_status() {
    # Check if nginx service is active (running)
    if systemctl is-active --quiet nginx; then
        NGINX_ACTIVE=true
        print_status "Nginx service is running"
    else
        NGINX_ACTIVE=false
        print_warning "Nginx service is not running"
    fi
    
    # Check if nginx service is enabled
    if systemctl is-enabled --quiet nginx; then
        NGINX_ENABLED=true
        print_status "Nginx service is enabled"
    else
        NGINX_ENABLED=false
        print_warning "Nginx service is not enabled"
    fi
}

# Function to start nginx service
start_nginx() {
    print_status "Starting nginx service..."
    sudo systemctl start nginx
    
    if [ $? -eq 0 ]; then
        print_status "Nginx service started successfully!"
    else
        print_error "Failed to start nginx service!"
        exit 1
    fi
}

# Function to enable nginx service
enable_nginx() {
    print_status "Enabling nginx service..."
    sudo systemctl enable nginx
    
    if [ $? -eq 0 ]; then
        print_status "Nginx service enabled successfully!"
    else
        print_error "Failed to enable nginx service!"
        exit 1
    fi
}

# Function to display nginx status
show_nginx_info() {
    echo ""
    print_status "=== Nginx Status Summary ==="
    echo "Version: $(nginx -v 2>&1 | cut -d' ' -f3)"
    echo "Service Status: $(systemctl is-active nginx)"
    echo "Service Enabled: $(systemctl is-enabled nginx)"
    echo "Listening on: $(ss -tlnp | grep nginx | awk '{print $4}' | head -1)"
    echo ""
    
    # Test nginx configuration
    print_status "Testing nginx configuration..."
    sudo nginx -t
}

# Main execution
main() {
    print_status "=== Nginx Management Script ==="
    print_status "Starting nginx installation and configuration check..."
    
    # Check if nginx is installed
    if ! check_nginx_installed; then
        install_nginx
    else
        print_status "Nginx is already installed"
    fi
    
    # Check nginx service status
    check_nginx_status
    
    # Start nginx if it's not running
    if [ "$NGINX_ACTIVE" = false ]; then
        start_nginx
    fi
    
    # Enable nginx if it's not enabled
    if [ "$NGINX_ENABLED" = false ]; then
        enable_nginx
    fi
    
    # Show final status
    show_nginx_info
    
    print_status "Nginx management completed successfully!"
    
    # Optional: Test if nginx is responding
    print_status "Testing nginx response..."
    if curl -s -o /dev/null -w "%{http_code}" http://localhost | grep -q "200\|403"; then
        print_status "Nginx is responding correctly!"
    else
        print_warning "Nginx may not be responding properly. Check the configuration."
    fi
}

# Run main function
main "$@"
```

## Exercise 2: Create cronjob

### Requirements

- Write script to check system usage
- system infomation, cpu, memory, disk
- config cronjob to run every 10 min and write output to file sysusage-<datetime>.txt
- make the script exutable `chmod +x <file>`
- Set cronjob schedule using `crontab -e`
```
*/10 * * * * script.sh

```

```shell
#!/bin/bash

# Configuration
LOG_DIR="$HOME/system-logs"
TIMESTAMP=$(date '+%Y-%m-%d_%H-%M-%S')
OUTPUT_FILE="$LOG_DIR/sysusage-$TIMESTAMP.txt"

# Create log directory if it doesn't exist
mkdir -p "$LOG_DIR"

# Function to get system information
get_system_info() {
    echo "=== SYSTEM INFORMATION ==="
    echo "Hostname: $(hostname)"
    echo "Date: $(date)"
    echo "Uptime: $(uptime -p)"
    echo "OS: $(cat /etc/os-release | grep PRETTY_NAME | cut -d'=' -f2 | tr -d '"')"
    echo ""
}

# Function to get CPU information
get_cpu_info() {
    echo "=== CPU INFORMATION ==="
    echo "CPU Model: $(grep 'model name' /proc/cpuinfo | head -1 | cut -d':' -f2 | xargs)"
    echo "CPU Cores: $(nproc)"
    
    # Simple CPU usage calculation
    CPU_USAGE=$(top -bn1 | grep "Cpu(s)" | awk '{print $2+$4}' | cut -d'%' -f1)
    echo "CPU Usage: ${CPU_USAGE}%"
    
    echo "Load Average: $(uptime | awk -F'load average:' '{print $2}')"
    echo ""
    
    echo "Top 3 CPU Processes:"
    ps aux --sort=-%cpu | head -4 | tail -3 | awk '{printf "  %-10s %s%%   %s\n", $1, $3, $11}'
    echo ""
}

# Function to get memory information
get_memory_info() {
    echo "=== MEMORY INFORMATION ==="
    
    # Get memory info in MB
    MEM_TOTAL=$(free -m | grep 'Mem:' | awk '{print $2}')
    MEM_USED=$(free -m | grep 'Mem:' | awk '{print $3}')
    MEM_FREE=$(free -m | grep 'Mem:' | awk '{print $4}')
    MEM_PERCENT=$((MEM_USED * 100 / MEM_TOTAL))
    
    echo "Total Memory: ${MEM_TOTAL} MB"
    echo "Used Memory: ${MEM_USED} MB"
    echo "Free Memory: ${MEM_FREE} MB"
    echo "Memory Usage: ${MEM_PERCENT}%"
    
    # Swap information
    SWAP_TOTAL=$(free -m | grep 'Swap:' | awk '{print $2}')
    SWAP_USED=$(free -m | grep 'Swap:' | awk '{print $3}')
    
    if [ "$SWAP_TOTAL" -gt 0 ]; then
        SWAP_PERCENT=$((SWAP_USED * 100 / SWAP_TOTAL))
        echo "Swap Total: ${SWAP_TOTAL} MB"
        echo "Swap Used: ${SWAP_USED} MB (${SWAP_PERCENT}%)"
    else
        echo "Swap: Not configured"
    fi
    echo ""
    
    echo "Top 3 Memory Processes:"
    ps aux --sort=-%mem | head -4 | tail -3 | awk '{printf "  %-10s %s%%   %s\n", $1, $4, $11}'
    echo ""
}

# Function to get disk information
get_disk_info() {
    echo "=== DISK INFORMATION ==="
    echo "Disk Usage:"
    df -h | grep -E '^/dev/' | awk '{printf "  %-15s %s used of %s (%s)\n", $6, $3, $2, $5}'
    echo ""
    
    echo "Largest Directories in /home:"
    du -sh /home/* 2>/dev/null | sort -hr | head -5 | awk '{printf "  %s\t%s\n", $1, $2}' || echo "  Cannot access /home directories"
    echo ""
}

# Function to generate the complete report
generate_report() {
    echo "System Monitor Report"
    echo "===================="
    echo ""
    
    get_system_info
    get_cpu_info
    get_memory_info
    get_disk_info
    
    echo "===================="
    echo "Report generated at: $(date)"
}

# Main execution
main() {
    generate_report > "$OUTPUT_FILE"
    echo "Report saved to: $OUTPUT_FILE"
}

# Run the script
main "$@"
```
